# Functions
INCLUDE(CheckFunctionExists)
MACRO(CHECK_FUNCTION_EXISTS_01 func var)
	CHECK_FUNCTION_EXISTS(${func} ${var})
	IF(NOT ${var})
		SET(${var} 0 CACHE INTERNAL "" FORCE)
	ENDIF(NOT ${var})
ENDMACRO(CHECK_FUNCTION_EXISTS_01)

CHECK_FUNCTION_EXISTS_01(getcwd HAVE_GETCWD)

INCLUDE(CheckIncludeFiles)
MACRO(CHECK_INCLUDE_FILES_01 incfile var)
	CHECK_INCLUDE_FILES(${incfile} ${var})
	IF(NOT ${var})
		SET(${var} 0 CACHE INTERNAL "" FORCE)
	ENDIF(NOT ${var})
ENDMACRO(CHECK_INCLUDE_FILES_01)
