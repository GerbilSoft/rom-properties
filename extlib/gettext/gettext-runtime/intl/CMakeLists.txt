# Functions
INCLUDE(CheckFunctionExists)
MACRO(CHECK_SYMBOL_EXISTS_01 func header var)
	CHECK_SYMBOL_EXISTS(${func} ${header} ${var})
	IF(NOT ${var})
		SET(${var} 0 CACHE INTERNAL "" FORCE)
	ENDIF(NOT ${var})
ENDMACRO(CHECK_SYMBOL_EXISTS_01)

MACRO(CHECK_INCLUDE_FILES_01 incfile var)
	CHECK_INCLUDE_FILES(${incfile} ${var})

ADD_LIBRARY(${PROJECT_NAME} SHARED
	${${PROJECT_NAME}_GL_SRCS}
	${${PROJECT_NAME}_SRCS}
	)
SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES
	VERSION ${${PROJECT_NAME}_VERSION}
	SOVERSION ${${PROJECT_NAME}_SOVERSION}
	)
TARGET_COMPILE_DEFINITIONS(${PROJECT_NAME}
	PRIVATE	BUILDING_LIBINTL
		#IN_LIBINTL
		ENABLE_RELOCATABLE=1
		IN_LIBRARY
		NO_XMALLOC
		set_relocation_prefix=libintl_set_relocation_prefix
		relocate=libintl_relocate
		HAVE_CONFIG_H
		LOCALEDIR=""
		BUILDING_LIBRARY=1
		DLL_EXPORT=1
	)
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME}
	PRIVATE	$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
		$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/gnulib-lib>
	PUBLIC	$<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}>
	)
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} SYSTEM
	PRIVATE $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}/gnulib-lib>
	)

# FIXME: MSVC doesn't seem to support the `restrict` keyword when building
# for Windows XP using v141_xp. (MSVC 2026 seems to accept it, but it isn't
# documented...)
IF(MSVC)
	TARGET_COMPILE_DEFINITIONS(${PROJECT_NAME}
		PRIVATE	restrict=
		)
ENDIF(MSVC)

DO_SPLIT_DEBUG(${PROJECT_NAME})
SET_WINDOWS_SUBSYSTEM(${PROJECT_NAME} WIN32)
#SET_WINDOWS_NO_MANIFEST(${PROJECT_NAME})

IF(WIN32)
	# Append the SOVERSION and "d" for debug.
	SET(CMAKE_DEBUG_POSTFIX "d")
	SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES DEBUG_POSTFIX "-${${PROJECT_NAME}_SOVERSION}d")
	SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES RELEASE_POSTFIX "-${${PROJECT_NAME}_SOVERSION}")
ENDIF(WIN32)

# FIXME: The libintl_* prefixed symbols aren't being exported properly.
# For MinGW-w64, the .def is added as a source file.
# For MSVC, we have to add a linker flag.
IF(MSVC)
	SET(DEF_FLAG "/DEF:${CMAKE_CURRENT_SOURCE_DIR}/gnuintl-${${PROJECT_NAME}_SOVERSION}.def")
	GET_TARGET_PROPERTY(RP_LINK_FLAGS ${PROJECT_NAME} LINK_FLAGS)
	IF(RP_LINK_FLAGS)
		SET(RP_LINK_FLAGS "${RP_LINK_FLAGS} ${DEF_FLAG}")
	ELSE()
		SET(RP_LINK_FLAGS "${DEF_FLAG}")
	ENDIF()
	SET_TARGET_PROPERTIES(${PROJECT_NAME} PROPERTIES LINK_FLAGS "${RP_LINK_FLAGS}")
ENDIF(MSVC)

######################################
# Install the library. (SHARED only) #
######################################

IF(WIN32)
	# NOTE: Don't install libraries.
	# That installs the import library, which we don't
	# want to distribute.
	INSTALL(TARGETS ${PROJECT_NAME}
		RUNTIME DESTINATION "${DIR_INSTALL_EXE}"
		LIBRARY DESTINATION "${DIR_INSTALL_DLL}"
		#ARCHIVE DESTINATION "${DIR_INSTALL_LIB}"
		COMPONENT "dll"
		)
ELSE(WIN32)
	INSTALL(TARGETS ${PROJECT_NAME}
		LIBRARY DESTINATION "${DIR_INSTALL_LIB}"
		COMPONENT "lib"
		)
ENDIF(WIN32)

# Check if a split debug file should be installed.
IF(INSTALL_DEBUG)
	# FIXME: Generator expression $<TARGET_PROPERTY:${_target},PDB> didn't work with CPack-3.6.1.
	GET_TARGET_PROPERTY(DEBUG_FILENAME ${PROJECT_NAME} PDB)
	IF(WIN32)
		# TODO: Test on MinGW.
		# FIXME: Generator PDB expression doesn't have the debug postfix for some reason,
		# so cpack fails in debug builds if we get the PDB property.
		SET(PDB_FILENAME_D "$<TARGET_FILE_DIR:${PROJECT_NAME}>/${PROJECT_NAME}-${${PROJECT_NAME}_SOVERSION}d.pdb")
		SET(PDB_FILENAME_R "$<TARGET_FILE_DIR:${PROJECT_NAME}>/${PROJECT_NAME}-${${PROJECT_NAME}_SOVERSION}.pdb")
		INSTALL(FILES "${PDB_FILENAME_D}" "${PDB_FILENAME_R}"
			DESTINATION "${DIR_INSTALL_DLL_DEBUG}"
			COMPONENT "debug"
			OPTIONAL
			)
		UNSET(PDB_FILENAME_D)
		UNSET(PDB_FILENAME_R)
	ELSE(WIN32)
		INSTALL(FILES "${DEBUG_FILENAME}"
			DESTINATION "${DIR_INSTALL_DLL_DEBUG}"
			COMPONENT "debug"
			)
	ENDIF(WIN32)
	UNSET(DEBUG_FILENAME)
ENDIF(INSTALL_DEBUG)
